# Codebase

## Setup một số script cơ bản trong package.json

```bash
{
 "scripts": {
    "nx:reset": "nx reset",
    "nx:run-many": "nx run-many",
    "lint": "pnpm run nx:run-many --target=lint",
    "fix": "pnpm run nx:run-many --target=lint --fix",
    "nx:affected": "nx affected --parallel=2 --nxBail",
    "docker-build": "pnpm nx:run-many --target=docker-build --verbose",
    "docker:start:provider": "docker compose -f docker-compose.provider.yaml up -d",
    "docker:start:dev": "docker compose -f docker-compose.dev.yaml up -d",
    "docker:stop:dev": "docker compose -f docker-compose.dev.yaml down",
    "build": "pnpm nx:reset && pnpm nx:run-many -t build --no-cache && pnpm docker-build && pnpm docker:start:provider && pnpm docker:stop:dev && pnpm docker:start:dev",
    "dev": "pnpm nx:reset && pnpm nx:run-many -t serve", -
    "dev-lite": "pnpm dev --projects=bff,authorizer,user-access,product,invoice",
    "prepare": "husky"
  },
}
```

Chú thích:

- `dev`: chạy toàn bộ các service trong workspace
- `dev-lite`: chạy các service chỉ định
